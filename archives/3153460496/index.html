排序

最常用的排序，快速排序，属于交换类排序。

void qSort(int a[], int l, int r) {
    if (l >= r) return; //递归结束条件 划分到只剩下一个数

    int x = a[l + r >> 1], i = l - 1, j = r + 1; //选定比较数的方法还有很多
    while (i < j) {
        do i++; while (a[i] < x);
        do j++; while (a[j] > x);
        if (i < j) swap(a[i], a[j]);
    }
    qSort(a, l, j), qSort(a, j + 1, r);
}

快排日常已经够用了，但也有不能满足需要的时候，比如求逆序对，这个时候就要用到归并排序。

void mergeSort(int a[], int l, int r) {
    if (l >= r) return; //递归结束条件 划分到只剩下一个数
    
    int mid = l + r >> 1;
    mergeSort(a, l, mid);
    mergeSort(a, mid + 1, r);
    int t[N];
    int index = 0, i = l, j = mid + 1;
    //对一分为二的数组 进行比较 然后拷贝到临时数组里面
    while (i <= mid && j <= r) {    
        if (a[i] < a[j]) t[index++] = a[i++];
        else t[index++] = a[j++];
    }
    //两个数组可能还有剩余 这时直接把剩下的全拷贝过去
    while (i <= mid) t[index++] = a[i++];
    while (j <= r) t[index++] = a[j++];
    //最后拷贝已经有序临时数组 
    for (i = l, j = 0; i <= r; i ++, j ++ ) a[i] = t[j];
}

二分查找

要求已有数据为有序状态。

整数二分和浮点数二分的区别是

  1. 循环结束条件的写法

  2. 浮点数不能使用位运算

//整数
while (l < r) {
     mid = (r - l) >> 1 + l;
}
//浮点数 
//因为浮点数会有精度损失问题 所以使用这样的比较写法
//1e-n 保留 n 位小数 保险起见可以多加一位
while (r - l < 1e-n) {
    mid = (r - l) / 2.0 + l;
}

划分的区间为 [l, mid]，[mid + 1, r] 时，一般用途求最小化。

//划分的区间为[l, mid]，[mid + 1, r]
int binarySearch_1(int l, int r) {
    while (l < r) {
        int mid = (r - l) >> 1 + l; //防止溢出 好习惯 从我做起
        if (check(mid)) r = mid;
        else l = mid + 1;
    }
    return l;
}

划分的区间为 [l, mid - 1]，[mid, r] 时，一般用途求最大化。

//划分的区间为[l, mid - 1]，[mid, r]
int binarySearch_1(int l, int r) {
    while (l < r) {
        int mid = (r - l + 1) >> 1 + l; //需要额外加1 防止死循环
        if (check(mid)) l = mid;
        else r = mid - 1;
    }
    return l;
}

如果你对区间的划分和对应的二分记不清楚的话，我这里提供一个我自己的小技巧

  1. 求最大化，右边比左边大，那 mid 就在右区间，而 mid 的左边 mid - 1 就是左区间，即[l, mid - 1]，[mid, r]；
  2. 求最小化，左边比右边小，那 mid 就在左区间，而 mid 的右边 mid + 1 就是右区间，即[l, mid]，[mid + 1, r]。
  3. 另外，如果只是单纯查找一个数，两个模板都可以用，或者使用最朴素的二分查找。

大数加减乘除

当运算数连 long long 都保存不了，就使用字符串或者整型数组来处理这类问题。

加法

//123 在vector中为321 逆序存放
//& 运算符可以加快传参速度
vector<int> add(vector<int> &A, vector<int> &B) {
    //采用大数 + 小数的方式
    if (A.size() < B.size()) return add(B, A);

    vector<int> ret;
    int carry = 0; //保存加法进位
    for (int i = 0; i < A.size(); i++) {
        //不断进行末位相加 结果是余数 进位不断前进
        carry += A[i];
        if (i < B.size()) carry += B[i];
        ret.push_back(carry % 10);
        carry /= 10;
    }
    //如果最后还有进位 就加进去
    if (carry) ret.push_back(carry);
    return ret;
}

减法

//与加法同理
vector<int> sub(vector<int> &A, vector<int> &B)
{
    vector<int> ret;
    int carry = 0;
    for (int i = 0, t = 0; i < A.size(); i++) {
        carry = A[i] - carry;//如果有借位 就减掉
        if (i < B.size()) carry -= B[i];
        ret.push_back((carry + 10) % 10);
        //每次借位只有0 或 1两种情况
        if (carry < 0) t = 1; 
        else carry = 0;
    }
    //消除前导0
    while (ret.size() > 1 && ret.back() == 0) ret.pop_back();
    return ret;
}

//首先是大数与一位数的乘法 再扩展到大数与大数的乘法
vector<int> mulSingle(vector<int> &A, int b) {
    vector<int> ret;
    int t = 0;
    
    for (int i = 0; i < A.size() || t; i++) {
        if (i < A.size()) t += A[i] * b;
        ret.push_back(t % 10);
        t /= 10;
    }
    //消除前导0 比如123 可能逆序存放为3210
    while (ret.size() > 1 && ret.back() == 0) ret.pop_back();

    return ret;
}

因数个数与因数之和

因数个数

//因数个数
//原理: 每个质因子的最高次方+1的连乘
long long cal(int x) {
    long long ans = 1, res = 0, temp = 0;
    Loop(i, 2, x) {
        res = 0;
        while (x % i == 0) {
            ++res;
            x /= i;
        }
        ans *= (res + 1);
    }
    return ans;
}

因数之和

//因数之和
//原理: 每个质因子从0次方加到它的最高次方，然后连乘。
//比如：求18 ，（2^0 + 2)*(3^0+3+3^2) = 3*13 = 39 
long long cal(int x) {
    long long ans = 1, res = 0, temp = 0;
    Loop(i, 2, x) {
        res = 1;
        while (x % i == 0) {
            res *= i;
            x /= i;
        }
        temp = (res * i - 1) / (i - 1);
        ans *= temp;
    }
    return ans;
}
