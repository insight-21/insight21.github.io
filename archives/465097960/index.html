AcWing 3250. 通信网络

又是一道很经典的图论题, dfs标记状态

DFS + 标记状态

使用一个st[i][j]数组, 表示 $i$ 到 $j$ 是否连通
建立好图之后, dfs跑一遍
每次记录当前点和父节点 标记两个点的状态为连通
两个点的关系只访问一次
最后检查一下状态数组中, 两个点是否连通即可
时间复杂度 $O(N^2)$

参考代码

#include<iostream>
#include<cstring>
using namespace std;

const int N = 1005, M = 1e5 + 5;
int h[N], to[M], nxt[M], idx;
bool st[N][N];

void add(int u, int v) {
    to[idx] = v;
    nxt[idx] = h[u];
    h[u] = idx++;
}
/*
每次记录当前点和父节点 标记两个点的状态为连通
两个点的关系只访问一次
*/
void dfs(int fa, int x) {
    if (st[fa][x]) return ;
    st[fa][x] = 1;
    for (int i = h[x]; ~i; i = nxt[i]) {
        int j = to[i];
        dfs(fa, j);
    }
}

int main() {
    memset(h, -1, sizeof h);
    
    int n, m; cin >> n >> m;
    for (int i = 0; i < m; ++i) {
        int a, b;
        cin >> a >> b;
        add(a, b);
    }
    
    for (int i = 1; i <= n; ++i) {
        dfs(i, i);
    }
    
    /* 检查两两之间是否连通 */
    int ans = 0;
    for (int i = 1; i <= n; ++i) {
        bool flag = true;
        for (int j = 1; j <= n; ++j) {
            if (!st[i][j] && !st[j][i]) {
                flag = false;
                break;
            }
        }
        ans += flag;
    }
    cout << ans;
    
    return 0;
}

AcWing 3215. 网络延时

经典图论板子题

最远点对

这道题本质上是求图中最远的两个点, 转到树上就变求最远的两个叶子节点, 最远的两个点必然是叶子节点
建图用邻接表, 在这里我使用链式前向星建立邻接表
然后dfs遍历邻接表, 每次求出当前点的最远和次远叶子节点距离
这两个点就是当前点能达到的最远点对, 两者之和就是最远距离
遍历的时候取最大值即可

参考代码

#include<iostream>
#include<cstring>
using namespace std;

const int N = 2e4 + 5;
int h[N], nxt[N], to[N], idx;
int ans;

void add(int u, int v) {
    to[idx] = v;
    nxt[idx] = h[u];
    h[u] = idx++;
}

int dfs(int u) {
    // 定义最远和次远叶节点距离
    int d1 = 0, d2 = 0;
    for (int i = h[u]; ~i; i = nxt[i]) {
        int j = to[i];
        int t = dfs(j);
        if (t >= d1) d2 = d1, d1 = t;
        else if (t > d2) d2 = t;
    }
    ans = max(ans, d1 + d2);
    
    return d1 + 1;
}

int main() {
    memset(h, -1, sizeof h);
    
    int n, m; cin >> n >> m;
    int len = n + m;
    
    // 因为根节点从1开始 所以输入直接从2开始
    for (int i = 2, x; i <= len; ++i) {
        cin >> x;
        add(x, i);
    }
    
    dfs(1);
    
    cout << ans << endl;
    
    return 0;
}

AcWing 3205. 最优配餐

开始做图论咯

多源BFS

可以以分店/客户坐标作为起点, 然后计算所有点到所有分店/客户的最小距离
最后客户到分店/分店到客户的距离 * 订单数即可
计算最短距离可以用bfs扩散, 快一点

参考代码

#include<iostream>
#include<cstring>
#include<queue>
using namespace std;

typedef pair<int, int> pii;

const int N = 1e3+5;
const int M = 1e6+5;
int customer[M][2], amount[M]; // 客户坐标 订单数
bool red[N][N]; // 不能通过的点
int dis[N][N]; //用来记录每个点到分店的最小距离
int n, m, k, d;
queue<pii> q;

void bfs() {
    int dxy[4][2] = {{-1, 0}, {0, -1}, {1, 0}, {0, 1}};
    
    while (q.size()) {
        auto t = q.front(); q.pop();
        
        for (int i = 0, tx, ty; i < 4; ++i) {
            tx = t.first + dxy[i][0];
            ty = t.second + dxy[i][1];
            // 注意数据是从1开始
            if (tx < 1 || tx > n || ty < 1 || ty > n || red[tx][ty]) continue;
            if (dis[tx][ty] > dis[t.first][t.second] + 1) {
                dis[tx][ty] = dis[t.first][t.second] + 1;
                q.push({tx, ty});
            }
        }
    }
}

int main() {
    // 输入数据 防止TLE
    ios::sync_with_stdio(0);
    cin.tie(0);
    
    cin >> n >> m >> k >> d;
    
    memset(dis, 0x3f, sizeof dis);
    // 把所有分店作为起点去bfs
    for (int i = 0, x, y; i < m; ++i) {
        cin >> x >> y;
        q.push({x, y});
        dis[x][y] = 0;
    }
    for (int i = 0; i < k; ++i) {
        cin >> customer[i][0] >> customer[i][1] >> amount[i];
    }
    for (int i = 0, x, y; i < d; ++i) {
        cin >> x >> y;
        red[x][y] = 1;
    }
    
    // bfs得到所有点的距离
    bfs();
    
    // 输出最小距离 * 订单数
    long long ans = 0;
    for (int i = 0; i < k; ++i) {
        ans += dis[customer[i][0]][customer[i][1]] * amount[i];
    }
    cout << ans;
    
    return 0;
}

AcWing 3195. 有趣的数

难得的动态规划

数位DP

按照规则
最高位不能为0, 因此第一位不会为0
0必须在1前面 所以第一位也不会为1
2必须在3前面 所以第一位不会为3
因此初始状态只能是2
确定初始状态后, 就能确定后续状态如下
0: 用了2, 还剩0, 1, 3, 初始状态
1: 用了2, 0, 还剩1, 3
2: 用了2, 3, 还剩0, 1
3: 用了2, 0, 1, 还剩3
4: 用了2, 0, 3, 还剩1
5: 用了2, 0, 3, 1
一共六种状态
其中状态转移为
0->0 0->1 0->2
1->1 1->3 1->4
2->2 2->4 2->5
3->3 3->5
4->4 4->5
5->5

绘制成图如下

比如, 指向状态 $0123$ 圈圈的箭头表示，$0123$ 状态只能由本身及 $012$ 和 $023$ 状态转移过来
时间复杂度 $O(N)$

参考代码

#include<iostream>
using namespace std;

typedef long long ll;

ll f[1005][6];
ll mod = 1e9 + 7;
ll a2, a02, a23, a012, a023, a0123;
ll t2, t02, t23, t012, t023, t0123;
int main() {
    int n; cin >> n;
    // 一开始每种状态的答案都为0 还没使用
    for (int i = 0; i < 6; ++i) f[0][i] = 0;
    
    for (int i = 1; i <= n; ++i) {
        // 乘以2 表示可以填两种
        ll j = i - 1;
        f[i][0] = 1;
        f[i][1] = (f[j][0] + f[j][1] * 2) % mod;
        f[i][2] = (f[j][0] + f[j][2]) % mod;
        f[i][3] = (f[j][1] + f[j][3] * 2) % mod;
        f[i][4] = (f[j][1] + f[j][2] + f[j][4] * 2) % mod;
        f[i][5] = (f[j][3] + f[j][4] + f[j][5] * 2) % mod;
    }
    cout << f[n][5];
    
    /* 优化做法 */
    // 由于每次只用到 i - 1 因此可以优化成滚动更新
    a2 = 1;
    for (int i = 1; i < n; ++i) {
        // 使用上一次 也就是 i - 1的数据
        t2 = a2, t02 = a02, t23 = a23, t012 = a012, t023 = a023, t0123 = a0123;
        a2 = t2 % mod;
        a02 = (t2 + t02 * 2) % mod;
        a23 = (t2 + t23) % mod;
        a012 = (t02 + t012 * 2) % mod;
        a023 = (t02 + t23 + t023 * 2) % mod;
        a0123 = (t012 + t023 + t0123 * 2) % mod;
    }
    cout << a0123;
    
    return 0;
}

AcWing 730. 机器人跳跃问题

一看到最小就想到二分, 其实这道题和上道题搭积木一样, 也可以反向思考

二分

二分值, 然后每次按照规则走一遍即可
不过这道题有个小细节, 如果你走到了序列的最大值, 按照规则, 你接下去只增不减, 肯定能到达终点, 这个时候直接返回即可
时间复杂度 $O(NlogN)$

参考代码

#include<iostream>
using namespace std;

const int N = 1e5+5;
int h[N];
int n, maxv;

bool check(int e) {
    for (int i = 0; i < n; ++i) {
        e += e - h[i];
        if (e >= maxv) return false;
        if (e < 0) return true;
    }
    return false;
}

int main() {
    /* 做法一 二分 */
    cin >> n;
    
    for (int i = 0; i < n; ++i) cin >> h[i], maxv = max(maxv, h[i]);

    int l = 0, r = 1e5;
    while (l < r) {
        int mid = l + r >> 1;
        if (check(mid)) l = mid + 1;
        else r = mid;
    }
    cout << l;
    
    return 0;
}

贪心(反向思考)

按照最优策略考虑, 如果走到终点, 能量刚好用完, 那一定是要求的最小值
所以逆序遍历, 按照规则反过来求最初值
注意题目要求向上取整, 对规则进行逆运算的时候, 用到了除法, 所以要对奇数偶数分别处理
时间复杂度 $O(N)$

参考代码

#include<iostream>
using namespace std;

const int N = 1e5+5;
int h[N], n;

int main() {    
    /* 做法二 贪心 */
    cin >> n;
    for (int i = 0; i < n; ++i) cin >> h[i];
    
    int init = 0;
    for (int i = n - 1, t; ~i; --i) {
        t = init + h[i];
        // 用位运算代替除法 奇数就加1 偶数不需要
        init = (t >> 1) + (t & 1);
    }
    cout << init;
    
    return 0;
}

AcWing 507. 积木大赛

差分的拓展题, 逆向思考

差分 + 贪心

区间操作, 很容易联想到差分处理
题目要求从0变到目标序列, 我们可以反向思考, 从目标序列变成0
也就是在区间上$[L, R]$减一, 转移到差分数组, 就是 $b[L] -= 1, b[R + 1] += 1$
又因为全0序列的差分序列也是全0, 也就是问你多少次区间操作能把差分序列变为0
既然要减一变为0, 对于差分数组中每个 $b_i > 0$, 至少需要减去 $b_i$次
所以总次数至少是差分序列中所有正数之和
那有没有办法可以做到操作次数一定是所有正数之和?
可以这样操作: 从后往前操作，如果当前的 $b_i>0$，则将其减1，并将其后的某个负数加1
证明如下:
由于差分数组的所有后缀 $\displaystyle\sum_{i=k}^{n}b_i = -h_k <= 0$
因此在所有后缀组合中, 正数的个数肯定小于负数, 所以总能找到对应的负数消除掉所有正数

参考代码

#include<iostream>
using namespace std;

const int N = 1e5 + 5;
int a[N];

int main() {
    int n; cin >> n;
    for (int i = 1; i <= n; ++i) cin >> a[i];
    
    int ans = 0;
    for (int i = n; i; --i) ans += max(0, a[i] - a[i - 1]);
    cout << ans; 
    
    return 0;
}

AcWing 148. 合并果子

拓展题: AcWing 282. 石子合并

哈夫曼树

理解题意, 就可以发现其实是哈夫曼树的模板题
哈夫曼树, 就是每次去所有数据里最小的两个, 合并成一个, 最后只剩下一个数据

参考代码

#include<iostream>
#include<queue>
#include<vector>

using namespace std;

int main() {
    int n; cin >> n;
    
    priority_queue<int, vector<int>, greater<int>> q;
    
    int t;
    while (n--) {
        cin >> t;
        q.push(t);
    }
    
    int ans = 0;
    while (q.size() > 1) {
        t = q.top(); q.pop();
        t += q.top(); q.pop();
        ans += t;
        q.push(t);
    }
    cout << ans;
    
    return 0;
}

AcWing 496. 机器翻译

一道模拟题 让我想起了万恶的面试算法LRU

队列 + 哈希

是否在内存, 用一个哈希标记即可
要清空最早的元素, 用队列
时间复杂度 $O(N)$
如果是Java, 直接反手一个LinkedHashMap(然后面试官就让你等通知

参考代码

#include<iostream>
#include<queue>

using namespace std;

bool cache[1005];
queue<int> q;
int ans;
int capacity, n;

/*
如果内存已存在 则直接返回
如果不存在 需要插入
插入后需要检查大小
*/
void put(int x) {
    if (cache[x]) return ;
    q.push(x);
    cache[x] = 1;
    ++ans;
    if (q.size() > capacity) {
        int t = q.front(); q.pop();
        cache[t] = 0;
    }
}

int main() {
    cin >> capacity >> n;
    
    for (int i = 0, t; i < n; ++i) {
        cin >> t;
        put(t);
    }
    cout << ans;
    
    return 0;
}

AcWing 211. 计算系数

是我不熟悉的数学题(doge

二项式定理 + 逆元

和这道题有关的有一个二项式定理
$(a+b)^k = \displaystyle\sum_{i=0}^{k}{C_k^i a^i b^{k-i}}$
通过这个定理, 我们可以求出题目要求的 $x^n, y^m$的系数就是 $C_k^n a^n b^m$
现在只需要想出怎么计算即可

  • $a^n \enspace b^m$ 由于题目给了模数 很容易想到 用快速幂解决
  • $C_k^n = \frac{k * (k-1) * … * (k - n + 1)} {1 * 2 * … * n}$ 涉及到除法取模 而乘法逆元就是为此而生的
  • 乘法逆元 由于模数$10007$是质数 因此可以借助费马小定理 即 $b$ 的逆元 $inv(b) = b^{mod - 2}$

    参考代码

    ```c++
    #include
    using namespace std;

int mod = 10007;

int ksm(int a, int b) {
int ret = 1;
a %= mod;
while (b) {
if (b & 1) ret = ret * a % mod;
a = a * a % mod;
b >>= 1;
}
return ret;
}

int main() {
int a, b, k, n, m;
cin >> a >> b >> k >> n >> m;

//C(k, n) * a^n * b^m
//a^n  b^m 使用快速幂
int ans = ksm(a, n) * ksm(b, m) % mod;

// C(k, n) = (k * k-1 * ... * 1) / (1 * 2 * ... * n);
// 一边乘一边除 因为mod=10007是质数 除法可以用乘法逆元
for (int i = 1, j = k; i <= n; ++i, --j) {
    ans = ans * j % mod;
    ans = ans * ksm(i, 10005) % mod;
}
cout << ans;

return 0;

}


## [AcWing 312. 乌龟棋](https://www.acwing.com/problem/content/314/)
说实话只能想到暴力, 因为数据也不大, 顶多 $40^4$
然后没想到这叫线性DP(?迷惑)

### 暴力/线性DP
直接对4种卡片的所有可能性进行枚举
可以利用动态规划的状态表示
用 $f[Card_A][Card_B][Card_C][Card_D]$ 来表示第 $i$ 种卡片使用了 $Card_i$ 张的最大分数
以使用一张 $A$ 卡为例, 分数的计算公式是 $f[Card_A - 1][Card_B][Card_C][Card_D] + score[A + 2 * B + 3 * C + 4 * D]$
时间复杂度 $O(40^4)$

### 参考代码
注意数组不要开太大, 4次方一下子就超出空间限制了
```c++
#include<iostream>
using namespace std;

const int N = 355, M = 45;
int f[M][M][M][M];
int score[N], b[5];

int main() {
    int n , m; cin >> n >> m;
    
    for(int i = 0; i < n; ++i) cin >> score[i];
    
    for (int i = 0, t; i < m; ++i) {
        cin >> t;
        ++b[t];
    }
    
    for (int A = 0; A <= b[1]; A ++ )
        for (int B = 0; B <= b[2]; B ++ )
            for (int C = 0; C <= b[3]; C ++ )
                for (int D = 0; D <= b[4]; D ++ ) {
                    int t = score[A + B * 2 + C * 3 + D * 4];
                    int &v = f[A][B][C][D];
                    v = t;
                    if (A) v = max(v, f[A - 1][B][C][D] + t);
                    if (B) v = max(v, f[A][B - 1][C][D] + t);
                    if (C) v = max(v, f[A][B][C - 1][D] + t);
                    if (D) v = max(v, f[A][B][C][D - 1] + t);
                }
    cout << f[b[1]][b[2]][b[3]][b[4]];
    
    return 0;
}

AcWing 1262. 鱼塘钓鱼

感觉是有动态规划解法的, 但是只能想到暴力枚举的做法, 顶多加个堆优化一下
待更新更好的解法

暴力枚举 + 堆

枚举只在前 $i$ 个鱼塘钓鱼, 然后求得到的钓鱼量, 然后取最大值
求得到的钓鱼量, 用总的钓鱼时间减去移动时间得到在钓鱼的时间, 然后枚举在钓鱼的每一分钟
最优策略是, 每分钟只去能得到最多鱼的池塘钓鱼, 最后把得到的答案更新
移动时间可以用前缀和处理, 快速得到前 $i$ 个鱼塘的移动时间
求每分钟最多鱼的鱼塘, 可以用堆

参考代码

#include<iostream>
#include<queue>

using namespace std;
typedef pair<int, int> pii;

const int N = 105;

//第一分钟鱼数量 每分钟减少量 去下一个鱼塘的时间
int a[N], d[N], s[N];

int main() {
    int T, n; cin >> n;
    for (int i = 1; i <= n; ++i) cin >> a[i];
    for (int i = 1; i <= n; ++i) cin >> d[i];
    //因为要快速求出 i 个鱼塘的移动时间 所以用前缀和
    for (int i = 2; i <= n; ++i) cin >> s[i], s[i] += s[i - 1];
    cin >> T;
    
    int ans = 0;
    //暴力枚举 只在前i个鱼塘钓鱼得到的鱼数量
    for (int i = 1; i <= n && s[i] <= T; ++i) {
        //因为每次要遍历这i个鱼塘 取得最大值
        //所以用堆来优化一下
        priority_queue<pii> q;
        for (int j = 1; j <= i; ++j) q.push({a[j], d[j]});
        //前i个鱼塘的鱼数量 以及除去移动时间的钓鱼总时间
        int tans = 0, t = T - s[i];
        //枚举在钓鱼的每一分钟
        for (int j = 1; j <= t; ++j) {
            auto p = q.top(); q.pop();
            if (p.first <= 0) break;
            tans += p.first; //每分钟的钓鱼量
            p.first -= p.second; //每分钟减少钓鱼量
            q.push(p);
        }
        ans = max(ans, tans);
    }
    cout << ans;
    
    return 0;
}

AcWing 257. 关押罪犯

把罪犯看作点, 怒气值看作边权, 就是图论题.

这题有两个做法, 是做完能得到很大收获的一道题.
染色二分图的判断模板: AcWing 860. 染色法判定二分图
并查集模板: Acwing 238. 银河英雄传说

排序 + 并查集

既然题目要求最小的怒气值, 那么干脆按照怒气值排序, 然后依次把两个罪犯分到两个监狱
如果做不到, 由于边权的有序性(?)，显然当前的怨气值就是答案中要求的最小怨气值
可以把放到同一个监狱这个操作, 看成并查集的合并
问题在于, 由于两个监狱两个罪犯, 会有两种合并可能

  1. $a$ 罪犯分到 $X$ 监狱, $b$ 罪犯分到 $Y$ 监狱
  2. $a$ 罪犯分到 $Y$ 监狱, $b$ 罪犯分到 $X$ 监狱

解决方法为, 先不急着合并, 只保存两者的相对关系, 表示两者不在同一个监狱
等到下次出现新的关系比如 $a$ 和 $c$, 因为最优策略是放到不同监狱, 也就是说 $c$ 会被放到和 $b$ 同一个监狱
这个时候把 $b$ 和 $c$ 合并即可
时间复杂度$O(NlogN)$, 此处 $N$ 为边数

参考代码

#include<iostream>
#include<algorithm>
using namespace std;

const int N = 2e4 + 5, M = 1e5 + 5;
int p[N]; // 并查集的父节点
int opp[N];

struct edge {
    int a, b, c;
    
    //重写排序规则
    bool operator < (const edge &o) const {
        return c > o.c;
    }
} e[M];

int find(int x) {
    return p[x] == x ? x : p[x] = find(p[x]);
}

void merge(int a, int b) {
    int x = find(a), y = find(b);
    p[x] = y;
}

int main() {
    int n, m; cin >> n >> m;
    
    for (int i = 0; i < m; ++i) {
        cin >> e[i].a >> e[i].b >> e[i].c;
    }
    
    sort(e, e + m);
    
    //初始化父节点数组
    for (int i = 1; i <= n; ++i) {
        p[i] = i;
    }
    
    for (int i = 0; i < m; ++i) {
        int a = find(e[i].a), b = find(e[i].b);
        //如果两个的父节点相同 即监狱相同 说明此时冲突
        if (a == b) {
            cout << e[i].c << endl;
            return 0;
        }
        
        //如果两者不存在相对关系
        if (!opp[a]) opp[a] = b;
        //如果有相对关系 则与相对的点合并
        else merge(opp[a], b);
        
        //同理
        if (!opp[b]) opp[b] = a;
        else merge(opp[b], a);
    }
    
    //没有则输出0
    cout << 0 << endl;
    return 0;
}

二分 + 染色二分图判断

分成两个监狱, 可以想到用二分图
要求最大值最小, 可以用二分怒气值, 如果当前怒气值可行, 那说明更小的也行, 缩小右区间
二分怒气值, 然后安排监狱也就是染色二分图, 如果二分图可以染色成功, 就是监狱分配成功, 反之失败.
时间复杂度$O((N+M)logC)$, 此处 $N$ 为点数, $M$ 为边数, $C$ 为最大怒气值

参考代码

#include<iostream>
#include<cstring>
using namespace std;

const int N = 2e4 + 5, M = 2e5 + 5; //需要开两倍... 不然会死循环
int h[N], to[M], nxt[M], w[M], cnt; //链式前向星 建立邻接表
int color[N]; //二分图染色数组
int n, m;

void add(int a, int b, int c) {
    w[cnt] = c;
    to[cnt] = b;
    nxt[cnt] = h[a];
    h[a] = cnt++;
}

bool dfs(int u, int c, int mid) {
    color[u] = c;
    for (int i = h[u]; ~i; i = nxt[i]) {
        //如果边小于等于mid 那么这俩点可以在一个监狱里
        //只有大于mid时才不能在监狱里 所以可以把小于等于mid的点跳过
        //相当于小剪枝
        if (w[i] <= mid) continue;
        int j = to[i];
        //如果已经染过色 就不用染了
        if (color[j]) {
            //如果颜色和上一个相同
            if (color[j] == c) return false;
        }
        //染下一个点
        else if (!dfs(j, 3 - c, mid)) return false;
    }
    return true;
}

bool check(int mid) {
    memset(color, 0, sizeof color);
    for (int i = 1; i <= n; ++i) {
        if (color[i] == 0) {
            if (!dfs(i, 1, mid)) return false;
        }
    }
    return true;
}

int main() {
    memset(h, -1, sizeof h);
    
    cin >> n >> m;
    
    
    int a, b, c;
    while(m--) {
        cin >> a >> b >> c;
        add(a, b, c);
        add(b, a, c);
    }
    
    int l = 0, r = 1e9 + 1;
    while (l < r) {
        int mid = l + r >> 1;
        if (check(mid)) r = mid;
        else l = mid + 1;
    }
    cout << l;
    
    return 0;
}

AcWing 503. 借教室

和 AcWing 499. 聪明的质监员 类似的一道题

二分 + 差分

随着天数推进，被借走的教室越来越多，剩余的教室越来越少，也就是单调递减
因此可以使用二分来找到教室不够用的那一天

接着要处理的就是快速求出每天教室的数量
由于是在天数的区间上借教室，也就是对区间进行常数运算
考虑使用差分进行借教室的操作，然后求和得出教室数量
差分模板：AcWing 797. 差分
$PS$: 前缀和的差分是原数列，差分的前缀和是原数列

时间复杂度 $O(klog(n))$

参考代码

#include<iostream>
using namespace std;

typedef long long ll;
const int N = 1e6 + 5;
int r[N], d[N], st[N], ed[N];
ll dif[N];
int n, m;

bool check(int mid) {
    //拷贝差分数组到临时数组
    for (int i = 1; i <= n; ++i) dif[i] = r[i];
    //差分数组的区间操作
    for (int i = 1; i <= mid; ++i) {
        dif[st[i]] -= d[i];
        dif[ed[i] + 1] += d[i];
    }
    //求和还原到原数列
    ll ret = 0;
    for (int i = 1; i <= n; ++i) {
        ret += dif[i];
        if (ret < 0) return false;
    }
    
    return true;
}

int main() {
    cin >> n >> m;
    for (int i = 1; i <= n; ++i) cin >> r[i];
    for (int i = 1; i <= m; ++i) {
        cin >> d[i] >> st[i] >> ed[i];
    }
    //处理成差分数组
    for (int i = n; i; --i) r[i] -= r[i - 1];
    
    int l = 1, r = m;
    while (l < r) {
        int mid = l + r >> 1;
        if (check(mid)) l = mid + 1;
        else r = mid;
    }
    
    if (check(l)) cout << 0;
    else cout << -1 << endl << l;
    
    return 0;
}

AcWing 499. 聪明的质监员

二分 + 前缀和

区间上的检验值 $Y_i$ = 区间矿石个数 * 价值之和，前提是$W_i$ >= $W$
也就是说 $W$ 越大 符合要求的矿石越少 自然 $Y_i$ 越小 则 $Y$ 越小
所以 $Y$ 和 $W$ 成反比，也就是单调递减
考虑使用二分找出最佳的 $W$

然后就是 $Y_i$ 的计算
为了快速求出区间上符合要求的个数以及对应的价值之和
用前缀和预处理 就可以快速得到结果

还有一个问题，当最后一次二分, 区间的长度最后为$2$时
有可能 $getY(l) > S$, $getY(r) < S$, 也有可能反过来，在这里取前者
此时需要对两个答案进行比较，才能确定哪个是最优解

我使用的二分是 $l = mid + 1, r = mid$
当区间长度为2时, $r = l + 1, mid = (l + r) / 2 = l$
最后肯定是 $getY(mid) = getY(l) > S$, 然后 $l = mid + 1 = r$, 结束二分
所以需要对结束二分前的 $mid$, 和结束后的 $mid + 1$ 进行比较
此时 $mid + 1 = l = r, mid = l - 1 = r - 1$
所以是对 $getY(l - 1)$ 和 $getY(l)$ 或者 $getY(r - 1)$ 和 $getY(r)$ 进行比较得到答案

$PS$: 如果你采用的是 $l = mid, r = mid - 1$, 可以自己进行推导, 此时需要比较的是 $getY(l)$ 和 $getY(l + 1)$

参考代码

#include<iostream>

using namespace std;
typedef long long ll;

const int N = 2e5 + 5;
int w[N], v[N], L[N], R[N], cnt[N];
ll S, sum[N];
int n, m;

ll getY(int mid) {
    for (int i = 1; i <= n; ++i) {
        if (w[i] >= mid) {
            sum[i] = sum[i - 1] + v[i];
            cnt[i] = cnt[i - 1] + 1;
        } else {
            sum[i] = sum[i - 1];
            cnt[i] = cnt[i - 1];
        }
    }
    
    ll ret = 0;
    for (int i = 0; i < m; ++i) {
        ret += (cnt[R[i]] - cnt[L[i] - 1]) * (sum[R[i]] - sum[L[i] - 1]);
    }
    
    return ret;
}

int main() {
    cin >> n >> m >> S;
    for (int i = 1; i <= n; ++i) cin >> w[i] >> v[i];
    for (int i = 0; i < m; ++i) cin >> L[i] >> R[i];
    
    int l = 0, r = 1e6 + 1;
    while (l < r) {
        int mid = (l + r) >> 1;
        if (getY(mid) > S) l = mid + 1;
        else r = mid;
    }
    
    cout << min(abs(getY(l) - S), abs(getY(l - 1) - S));
    
    return 0;
}

AcWing 125. 耍杂技的牛

与 AcWing 114. 国王游戏 相同套路的一道题

排序 + 贪心

时间复杂度 $O(nlog(n))$
做法：按照牛的重量与强壮程度之和 $W_i + S_i$ 排序，得到的便是最优做法。

证明：
第 $i$ 头牛的危险程度： $\displaystyle\sum_{j=1}^{i - 1}{W_j} - S_i$

接下来列出两头牛在位置顺序交换前后的危险程度

牛 交换前 交换后
$i$ $\displaystyle\sum_{j=1}^{i - 1}{W_j} - S_i$ $\displaystyle\sum_{j=1}^{i - 1}{W_j} + W_{i+1} - S_i$
$i + 1$ $\displaystyle\sum_{j=1}^{i}{W_j} - S_{i+1}$ $\displaystyle\sum_{j=1}^{i-1}{W_j} - S_{i+1}$

化简式子，减去$\displaystyle\sum_{j=1}^{i-1}{W_j}$

牛 交换前 交换后
$i$ $- S_i (a)$ $W_{i+1} - S_i (b)$
$i + 1$ $W_i - S_{i+1} (c)$ $- S_{i+1} (d)$

首先，明显 $a < b, d < c$
所以如果能得出 $b$ 和 $c$ 的关系，就能比较交换前后危险程度的大小
当 $W_{i+1} - S_i$ >= $W_i - S_{i+1}$，移项变成$W_{i+1} + S_{i+1}$ >= $W_i + S_i$，交换前的危险程度更小
当 $W_{i+1} - S_i$ < $W_i - S_{i+1}$，移项变成$W_{i+1} + S_{i+1}$ < $W_i + S_i$，交换前后的危险程度更小

可知交换前后的危险程度和 $W_i + S_i$ 有关，到此证毕。

具体做法：根据 $W_i + S_i$ 排序，然后枚举记录更大的危险程度，输出即可。

参考代码

#include<iostream>
#include<algorithm>
using namespace std;

typedef pair<int, int> pii;

const int N = 5e4 + 5;
pii a[N];

int main() {
    int n; cin >> n;
    int w, s;
    for (int i = 0; i < n; ++i) {
        cin >> w >> s;
        //因为sort函数是按照数组第一维来排序的
        //为了方便排序 直接把数组第一维变成两者之和
        a[i] = {w + s, w};
    }
    
    sort(a, a + n);
    
    int ans = -0x3f3f3f3f, sum = 0;
    for (int i = 0; i < n; ++i) {
        int w = a[i].second, s = a[i].first - w;
        //记录更大的危险程度 重量减去强壮程度
        ans = max(ans, sum - s);
        //计算重量之和
        sum += w;
    }
    
    cout << ans;
    
    return 0;
}
